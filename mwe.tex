\NewDocumentCommand{\scorebox}{ O{} m }
{
    \group_begin:
    \setcounter{questionnum}{0}
    \keys_set:nn { scorebox } { #1 }
    \keys_set:nn { scorebox } { question-num = #2 }
    \tl_clear:N \l__scorebox_tblr_body_tl
    \tl_build_begin:N \l__scorebox_tblr_body_tl
    \tl_build_put_right:Nn \l__scorebox_tblr_body_tl { \l__scorebox_question_str_tl }
    \prg_replicate:nn { \l__scorebox_question_num_int } {
        \tl_build_put_right:Nn \l__scorebox_tblr_body_tl { & \stepquestionnum }
    }
    \tl_build_put_right:Nn \l__scorebox_tblr_body_tl { & \l__scorebox_totalscore_str_tl }
    \tl_build_put_right:Nn \l__scorebox_tblr_body_tl {\\}
    \tl_build_put_right:Nn \l__scorebox_tblr_body_tl { \l__scorebox_score_str_tl }
    \prg_replicate:nn { \l__scorebox_question_num_int } {%
        \tl_build_put_right:Nn \l__scorebox_tblr_body_tl { & }%
    }%
        \tl_build_put_right:Nn \l__scorebox_tblr_body_tl { & }
    \tl_build_put_right:Nn \l__scorebox_tblr_body_tl {\\}
    \tl_build_end:N \l__scorebox_tblr_body_tl
    \bigskip
    \noindent\centering
    \begin{tblr}[%
        expand=\l__scorebox_tblr_body_tl%
        ]{%
        hlines,vlines,%
        width=.85\linewidth,%
        stretch=0,%
        rowspec={*{2}{Q[m,ht=\l__scorebox_scorebox_height_dim]}},
        colspec={%
            Q[c,m,wd=\l__scorebox_firstcolumn_width_dim]%
            *{\l__scorebox_question_num_int}{X[c,m]}%
            Q[c,m,wd=\l__scorebox_totalscore_width_dim]%
        }%
    }%
    \l__scorebox_tblr_body_tl%
    \end{tblr}
    \par
    \bigskip    
    \group_end:
}

\newlength{\gapinsert}
\newcommand{\insertgap}[1]{%
  \par
  \ifdim\dimexpr\pagegoal-\pagetotal<#1
    % Capture space remaining after page break
    \setlength{\gapinsert}{\dimexpr#1-\pagegoal+\pagetotal}%
    \pagebreak% Insert page break
    \vspace*{\gapinsert}% Insert additional/remaining gap; https://tex.stackexchange.com/q/33370/5764
  \else
    % Enough space available on page
    \vspace{#1}% Insert regular vspace
  \fi
}
